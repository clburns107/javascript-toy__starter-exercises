Relational Equality Operators

console.log( "5 == '5' is " + (5 == '5') ); //true correct
This expression says is the number '5' equal to the string '5'.

console.log( "5 == 'five' is " + (5 == 'five') ); //false correct
This expression says is the number '5' equal to the string 'five'.

console.log( "5 == '6' is " + (5 == '6') ); //false correct
This expression says is the number '5' equal to the number '6'.

console.log( "5 === '5' is " + (5 === '5') ); //false correct
This expression says is the number '5' is exactly equal to the string '5'.

console.log( "5 == 2 + 3 is " + (5 == 2 + 3) ); //truecorrect
This expression says the number '5' is equal to the numeric value of '2 + 3'.

console.log( "5 == 5 is " + (5 == 5) ); //true correct
This expression says the number '5' is equal to the number '5'

console.log( "5 === 5 is " + (5 === 5) ); //true correct
This expression says the number '5' is exactly equal to the number '5'.

console.log( "true == 'true' is " + (true == 'true') ); //true incorrect
This expression says the boolean 'true' is equal to the string 'true'.
This is actually false because javascript cannot convert a string to a boolean.

console.log( "true === 'true' is " + (true === 'true') ); //false correct
This expression says the boolean 'true' is exactly equal to the string 'true'.

console.log( "true == false is " + (true == false) ); //false correct
This expression says the boolean 'true' is equal to the boolean 'false'.

console.log( "5 == 12 is " + (5 == 12) ); //false correct
This expression says the number '5' is equal to the number '12'.

console.log( "5 != 12 is " + (5 != 12) ); //true correct
This expression says the number '5' is not equal to the number '12'.

console.log( "5 < 12 is " + (5 < 12) ); //true correct
This expression says the number '5' is less than the number '12'.

console.log( "5 <= 12 is " + (5 <= 12) ); //true correct
This expression says the number '5' is less than OR equal to the number '12'.

console.log( "5 > 12 is " + (5 > 12)); //false correct
This expression says the number '5' is greater than the number '12'.

console.log( "5 >= 12 is " + (5 >= 12)); //false correct
This expression says the number '5' is greater than OR equal to the number '12'.
